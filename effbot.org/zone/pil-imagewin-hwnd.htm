Using ImageWin With Tkinter

November 25, 2002 | Fredrik Lundh

# display image in Tkinter window using the ImageWin.Dib interface
# (requires PIL 1.1.4b2 or later)

from Tkinter import *
from PIL import Image, ImageWin

im = Image.open("Images/lena.jpg")

frame = Frame(bg="", width=im.size[0], height=im.size[1])
frame.pack()

def expose(event):
    dib = ImageWin.Dib(im)
    dib.expose(ImageWin.HWND(frame.winfo_id()))

frame.bind("", expose)

frame.mainloop()
